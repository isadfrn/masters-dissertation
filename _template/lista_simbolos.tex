\begin{simbolos}
  \item[$ \Gamma $] Letra grega Gama
  \item[$ \Lambda $] Letra grega Lambda
  \item[$ \zeta $] Letra grega minúscula zeta
  \item[$ \in $] Pertence
  \item[$ \subset $] Está contido em (subconjunto)
  \item[$ \epsilon $] Taxa de evasão (perda de participantes entre início e fim de um módulo)
  \item[$ \gamma $] Eficiência de engajamento gamificado (razão entre nota prática e pontos de experiência total)
  \item[$ P $] Conjunto total de participantes
  \item[$ P_f $] Subconjunto de participantes do gênero feminino
  \item[$ P_m $] Subconjunto de participantes do gênero masculino
  \item[$ N_i $] Número de inscritos no início da fase
  \item[$ N_c $] Número de concluintes aprovados
  \item[$ Nota_{pratica} $] Nota obtida nas atividades práticas (laboratórios e simuladores)
  \item[$ XP_{total} $] Total de pontos de experiência acumulados
  \item[$ g $] Gênero (variável independente)
  \item[$ f $] Fase do funil (variável independente)
  \item[$ S_{ae} $] Score de autoeficácia (média do questionário, escala Likert)
  \item[$ R $] Ranking (posição na lista classificatória)
  \item[$ D_t $] Desempenho técnico (notas em laboratórios e simuladores)
  \item[$ T_1 $] Transição Nivelamento $\rightarrow$ Básico
  \item[$ T_2 $] Transição Básico $\rightarrow$ Fundamental
  \item[$ T_3 $] Transição Fundamental $\rightarrow$ Especializado
  \item[$ T_4 $] Transição Especializado $\rightarrow$ Residência Tecnológica
\end{simbolos}
